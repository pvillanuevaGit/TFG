create schema neveritas;
use neveritas;

create table usuarios(
	id_usuario int NOT NULL AUTO_INCREMENT,
    usuario varchar(255) NOT NULL,
    password varchar(18) NOT NULL,
    PRIMARY KEY (id_usuario)
);

create table videos(
